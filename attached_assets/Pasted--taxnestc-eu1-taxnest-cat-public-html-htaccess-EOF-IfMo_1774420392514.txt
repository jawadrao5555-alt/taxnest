<IfModule mod_rewrite.c>
    RewriteEngine On
    RewriteBase /

    # Stop rewrite loops for already-rewritten paths
    RewriteRule ^taxnest/public/ - [L]

    # Serve static files from taxnest/public if they exist
    RewriteCond %{DOCUMENT_ROOT}/taxnest/public%{REQUEST_URI} -f
    RewriteRule ^(.+)$ /taxnest/public/$1 [L]

    # Everything else goes to Laravel via custom index.php
    RewriteCond %{REQUEST_FILENAME} !-f
    RewriteRule ^ index.php [L]
</IfModule>
EOF
